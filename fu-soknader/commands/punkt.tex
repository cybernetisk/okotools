    \newcommand{\raisedrulefill}[2][0ex]{\leaders\hbox{\rule[#1]{1pt}{#2}}\hfill}
    %%% ... \section %%%
    \titleformat{\section}{\Large\bfseries}{\thesection. }{0em}{#1\,\raisedrulefill[0.4ex]{1pt}}
    %%% ... \subsection %%%
    \titleformat{\subsection}{\bfseries}{\thesubsection. }{0em}{#1\,\raisedrulefill[0.4ex]{0.5pt}}
    %%% ... \subsubsection %%%
    \titleformat{\subsubsection}{\bfseries}{\thesubsubsection. }{0em}{#1\,\raisedrulefill[0.4ex]{0.5pt}}

%%% Adjusting spacing around section titles %%%
    %\titlespacing*{<command>}{<left>}{<before-sep>}{<after-sep>}
    \titlespacing*{\section}{0cm}{0.5cm}{0.3cm}
    \titlespacing*{\subsection}{0cm}{3cm}{0.3cm} %denne funker ikke(!)
    \titlespacing*{\subsubsection}{0cm}{0.5cm}{0.5cm}

    
%%% Making of the commands punkt (section), upunkt (subsection), uupunkt (subsubsection) %%%
\makeatletter
    %%% Punkt %%%
        \newcommand*{\punkt}{%
          \@ifstar{\@punkt@star}{\@punkt}%
        }
        \newcommand*{\@punkt}{%
          \@dblarg\@punkt@nostar%
        }
        \newcommand*{\@punkt@star}[2]{%
          \section*{#1}{#2}\vspace{10pt}%
        }
        %%% Needs revision: %%%
        \def\@punkt@nostar[#1]#2{%
          \section[{#1}]{#2}%
        }
    %%% Under- punkt %%%
        \newcommand*{\upunkt}{%
          \@ifstar{\@upunkt@star}{\@upunkt}%
        }
        \newcommand*{\@upunkt}{%
          \@dblarg\@upunkt@nostar%
        }
        \newcommand*{\@upunkt@star}[1]{%
          \subsection*{#1}%
        }
        \def\@upunkt@nostar[#1]#2{%
          \subsection[{#1}]{#2}%
        }
    %%% Under- under- punkt %%%
        \newcommand*{\uupunkt}{%
          \@ifstar{\@uupunkt@star}{\@uupunkt}%
        }
        \newcommand*{\@uupunkt}{%
          \@dblarg\@uupunkt@nostar%
        }
        \newcommand*{\@uupunkt@star}[1]{%
          \subsubsection*{#1}%
        }
        \def\@uupunkt@nostar[#1]#2{%
          \subsubsection[{#1}]{#2}%
        }
\makeatother
